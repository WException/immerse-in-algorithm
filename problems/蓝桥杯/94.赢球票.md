# 94.赢球票

蓝桥杯题库 [题目链接](https://www.lanqiao.cn/problems/94/learning/)。

> 2016，模拟，国赛，枚举，搜索

数据量小，枚举所有起始点，然后模拟过程即可。

模拟过程维护三个变量：
1. sum 当前的球票总数
2. num 当前数到的数
3. idx 起始点

### C++代码
```cpp
#include <iostream>
#include <vector>
using namespace std;

int main() {
    int n;
    cin >> n;
    int ans = 0;
    vector<int> list;
    
    for (int i = 0; i < n; i++) {
        int num;
        cin >> num;
        list.push_back(num);
    }

    for (int i = 0; i < n; i++) {
        vector<int> temp = list;
        int sum = 0;
        int num = 1;
        int idx = i;
        
        while (num <= n && !temp.empty()) {
            if (idx == temp.size()) {
                idx = 0;
            }
            
            if (temp[idx] == num) {
                sum += temp[idx];
                temp.erase(temp.begin() + idx);
                num = 1;
            } else {
                idx++;
                num++;
            }
        }
        ans = max(ans, sum);
    }
    cout << ans << endl;
    return 0;
}
```

### Java代码
```Java
import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner scan = new Scanner(System.in);
        int n = scan.nextInt();
        int ans = 0;
        ArrayList<Integer> list = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            list.add(scan.nextInt());
        }

        for (int i = 0; i <n; i++) {
            ArrayList<Integer> temp = new ArrayList<>(list);
            int sum = 0;
            int num = 1;
            int idx = i;
            while(num <= n && temp.size() != 0) {
                if(temp.get(idx) == num) {
                    sum += temp.remove(idx);
                    num = 1;
                }else {
                    idx++;
                    num++;
                }
                if (idx == temp.size()) {
                    idx = 0;
                }
            }
            ans = Math.max(ans, sum);
        }
        System.out.println(ans);
        scan.close();
    }
}
```

### Python3代码
```python
def main():
    n = int(input().strip())
    arr = list(map(int, input().split()))
    ans = 0
    
    for i in range(n):
        temp = arr.copy()
        total = 0
        num = 1
        idx = i
        
        while num <= n and temp:
            if idx == len(temp):
                idx = 0
                
            if temp[idx] == num:
                total += temp.pop(idx)
                num = 1
            else:
                idx += 1
                num += 1
                
        ans = max(ans, total)
    
    print(ans)

if __name__ == "__main__":
    main()
```
